  return {
        r: f(5),
        g: f(3),
        b: f(1)}
}

// input: r,g,b in [0,255], out: h in [0,360) and s,v in [0,1]
function rgb2hsv(r,g,b) {
  let v=Math.max(r,g,b), c=v-Math.min(r,g,b);
  let h= c && ((v==r) ? (g-b)/c : ((v==g) ? 2+(b-r)/c : 4+(r-g)/c)); 
  return {
        h: 60*(h<0?h+6:h),
        s: v&&c/v,
        v: v}
}
